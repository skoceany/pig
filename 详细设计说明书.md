```javascript
// ImageUploader.vue 核心方法
methods: {
  handleFileChange(event) {
    const file = event.target.files[0]
    if (file) {
      this.validateFile(file)
    }
  },
  validateFile(file) {
    const types = ['image/jpeg', 'image/png', 'image/bmp']
    if (!types.includes(file.type)) {
      this.$emit('error', '仅支持JPG、PNG、BMP格式')
      return
    }
    if (file.size > this.maxSize * 1024 * 1024) {
      this.$emit('error', `文件大小不能超过${this.maxSize}MB`)
      return
    }
    this.$emit('upload', file)
  }
}
```

#### 3.5.1 DiagnosisService

```python
# app/services/diagnosis_service.py
from pathlib import Path
from datetime import datetime
from app.models.pig_disease_model import PigDiseaseModel
from app.models.database import DiagnosisHistory
from app.utils.image_utils import save_image, preprocess_image
from app.config.settings import Settings

class DiagnosisService:
    def __init__(self):
        self.model = PigDiseaseModel()
        self.settings = Settings()
        
    async def diagnose(self, image_file):
        """
        执行诊断流程
        
        Args:
            image_file: 上传的图片文件
            
        Returns:
            诊断结果字典
        """
        # 保存图片
        image_path = await self._save_uploaded_image(image_file)
        
        # 预处理图片
        processed_image = preprocess_image(image_path)
        
        # 调用模型进行诊断
        prediction = self.model.predict(processed_image)
        
        # 生成疾病解读
        interpretation = self.model.generate_interpretation(
            prediction['disease'], 
            processed_image
        )
        
        # 构建结果
        result = {
            'is_sick': prediction['confidence'] > self.settings.confidence_threshold,
            'disease_name': prediction['disease'],
            'confidence': prediction['confidence'],
            'interpretation': interpretation['description'],
            'symptoms': interpretation['symptoms'],
            'recommendations': interpretation['recommendations']
        }
        
        # 保存诊断历史
        await self._save_history(image_path, result)
        
        return result
    
    async def _save_uploaded_image(self, image_file):
        """保存上传的图片到指定目录"""
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        filename = f'{timestamp}_{image_file.filename}'
        save_path = Path(self.settings.upload_dir) / filename
        await save_image(image_file, save_path)
        return str(save_path)
    
    async def _save_history(self, image_path, result):
        """保存诊断记录到数据库"""
        history = DiagnosisHistory(
            image_path=image_path,
            disease_name=result['disease_name'],
            confidence=result['confidence'],
            interpretation=result['interpretation'],
            symptoms=str(result['symptoms']),
            recommendations=str(result['recommendations'])
        )
        # 添加到数据库会话并提交
        from app.config.database import get_session
        async with get_session() as session:
            session.add(history)
            await session.commit()
```

### 4.2 模型类设计

```python
# app/models/pig_disease_model.py
import torch
import json
from transformers import AutoProcessor, AutoModelForVisionAndLanguageGeneration

class PigDiseaseModel:
    def __init__(self):
        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.processor = AutoProcessor.from_pretrained("LLaVA-7B")
        self.model = AutoModelForVisionAndLanguageGeneration.from_pretrained(
            "LLaVA-7B",
            torch_dtype=torch.float16 if torch.cuda.is_available() else torch.float32
        ).to(self.device)
        
        # 疾病名称列表
        self.disease_names = [
            '非洲猪瘟', '猪瘟', '猪口蹄疫', '猪繁殖与呼吸综合征',
            '猪圆环病毒病', '猪传染性胃肠炎', '猪伪狂犬病', '猪链球菌病',
            '副猪嗜血杆菌病', '猪丹毒', '猪传染性胸膜肺炎', '猪附红细胞体病'
        ]
        
    def predict(self, image):
        """
        使用多模态模型进行疾病预测
        
        Args:
            image: 预处理后的图像张量
            
        Returns:
            预测结果字典，包含疾病名称和置信度
        """
        # 构建提示词
        prompt = self._build_diagnosis_prompt()
        
        # 预处理输入
        inputs = self.processor(
            text=prompt,
            images=image,
            return_tensors="pt"
        ).to(self.device)
        
        # 推理
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=200,
                num_beams=5,
                temperature=0.7
            )
        
        # 解析结果
        result = self.processor.decode(outputs[0], skip_special_tokens=True)
        return self._parse_prediction(result)
    
    def generate_interpretation(self, disease_name, image):
        """
        生成疾病详细解读
        
        Args:
            disease_name: 疾病名称
            image: 图像特征
            
        Returns:
            解读结果字典
        """
        # 构建解读提示词
        prompt = self._build_interpretation_prompt(disease_name)
        
        # 生成解读
        inputs = self.processor(
            text=prompt,
            images=image,
            return_tensors="pt"
        ).to(self.device)
        
        with torch.no_grad():
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=500,
                num_beams=5,
                temperature=0.7
            )
        
        result = self.processor.decode(outputs[0], skip_special_tokens=True)
        return self._parse_interpretation(result)
    
    def _build_diagnosis_prompt(self):
        """构建诊断提示词"""
        diseases = ", ".join(self.disease_names)
        return f"""分析这张猪的照片，判断猪是否患病。如果患病，请从以下疾病中选择最可能的一种：
{diseases}
请只返回疾病名称和置信度，格式如下：
疾病名称: [疾病名]
置信度: [0-1之间的数值]
"""
    
    def _build_interpretation_prompt(self, disease_name):
        """构建疾病解读提示词"""
        return f"""请详细解读{disease_name}，包括以下方面：
1. 疾病概述
2. 主要症状
3. 传播途径
4. 治疗方法
5. 预防措施
请以JSON格式返回结果。
"""
    
    def _parse_prediction(self, text):
        """解析预测结果"""
        disease_name = "未知"
        confidence = 0.0
        
        for disease in self.disease_names:
            if disease in text:
                disease_name = disease
                break
        
        # 提取置信度
        if "置信度" in text:
            try:
                start = text.index("置信度") + 4
                end = text.index("\n", start) if "\n" in text[start:] else len(text)
                confidence = float(text[start:end].strip())
            except:
                confidence = 0.5
        
        return {"disease": disease_name, "confidence": confidence}
    
    def _parse_interpretation(self, text):
        """解析疾病解读"""
        try:
            # 尝试提取JSON部分
            start = text.index("{")
            end = text.index("}") + 1
            json_str = text[start:end]
            return json.loads(json_str)
        except:
            # 如果不是JSON格式，返回默认结构
            return {
                "description": text,
                "symptoms": [],
                "recommendations": []
            }
```

```python
# app/utils/image_utils.py
from PIL import Image
import torch
from torchvision import transforms

def preprocess_image(image_path, target_size=(224, 224)):
    """
    预处理图像，统一尺寸和格式
    
    Args:
        image_path: 图像文件路径
        target_size: 目标尺寸
        
    Returns:
        预处理后的图像张量
    """
    transform = transforms.Compose([
        transforms.Resize(target_size),
        transforms.CenterCrop(target_size),
        transforms.ToTensor(),
        transforms.Normalize(
            mean=[0.485, 0.456, 0.406],
            std=[0.229, 0.224, 0.225]
        )
    ])
    
    image = Image.open(image_path).convert('RGB')
    return transform(image).unsqueeze(0)

async def save_image(image_file, save_path):
    """
    保存上传的图片
    
    Args:
        image_file: 上传的文件对象
        save_path: 保存路径
    """
    content = await image_file.read()
    with open(save_path, 'wb') as f:
        f.write(content)
```

```python
def augment_image(image):
    """
    数据增强，用于训练阶段
    
    Args:
        image: PIL图像对象
        
    Returns:
        增强后的图像列表
    """
    transforms_list = [
        transforms.RandomHorizontalFlip(p=0.5),
        transforms.RandomRotation(degrees=15),
        transforms.RandomResizedCrop(size=(224, 224), scale=(0.8, 1.0)),
        transforms.ColorJitter(brightness=0.2, contrast=0.2, saturation=0.2, hue=0.1),
        transforms.RandomAffine(degrees=0, translate=(0.1, 0.1))
    ]
    
    augmented_images = []
    for transform in transforms_list:
        augmented_images.append(transform(image))
    
    return augmented_images
```

## 9. 数据加载工具

```python
# app/utils/data_loader.py
import os
import json

def load_disease_descriptions(data_dir="./多模态数据/train-data"):
    """
    加载所有疾病的描述文件
    
    Args:
        data_dir: 训练数据目录
        
    Returns:
        疾病信息列表
    """
    diseases = []
    disease_dirs = [d for d in os.listdir(data_dir) if os.path.isdir(os.path.join(data_dir, d))]
    
    for disease_name in disease_dirs:
        description_path = os.path.join(data_dir, disease_name, "描述.txt")
        if os.path.exists(description_path):
            with open(description_path, 'r', encoding='utf-8') as f:
                content = f.read()
                
            # 解析描述文件
            disease_info = parse_description_file(disease_name, content)
            diseases.append(disease_info)
    
    return diseases

def parse_description_file(disease_name, content):
    """
    解析单个疾病的描述文件
    
    Args:
        disease_name: 疾病名称
        content: 文件内容
        
    Returns:
        解析后的疾病信息字典
    """
    lines = content.strip().split('\n')
    description = []
    symptoms = []
    image_descriptions = []
    
    i = 0
    while i < len(lines):
        line = lines[i].strip()
        
        # 跳过空行和疾病名称行
        if not line or line == disease_name:
            i += 1
            continue
        
        # 检查是否是图片文件名
        if line.endswith('.jpg'):
            # 获取图片描述
            if i + 1 < len(lines) and not lines[i+1].strip().endswith('.jpg'):
                image_descriptions.append(lines[i+1].strip())
                # 从图片描述中提取症状
                symptoms.extend(extract_symptoms(lines[i+1].strip()))
            i += 2
            continue
        
        # 疾病描述文本
        description.append(line)
        i += 1
    
    return {
        'disease_name': disease_name,
        'description': '\n'.join(description),
        'symptoms': json.dumps(list(set(symptoms))),
        'transmission': '',
        'treatment': '',
        'prevention': '',
        'images_count': len(image_descriptions)
    }

def extract_symptoms(text):
    """
    从文本中提取症状关键词
    
    Args:
        text: 描述文本
        
    Returns:
        症状列表
    """
    symptom_keywords = [
        '出血', '发绀', '水泡', '肿胀', '红斑', '疹块', '溃疡',
        '发烧', '高烧', '腹泻', '呕吐', '瘫痪', '坏死', '充血',
        '水肿', '呼吸困难', '精神沉郁', '食欲不振'
    ]
    
    symptoms = []
    for keyword in symptom_keywords:
        if keyword in text:
            symptoms.append(keyword)
    
    return symptoms
```
